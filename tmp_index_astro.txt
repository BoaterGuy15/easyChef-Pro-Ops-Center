<script>
  // ── Email CTA form: stub submit handler ──────────────────────────────────────
  const form = document.querySelector<HTMLFormElement>('[data-cta-form]');
  if (form) {
    form.addEventListener('submit', (e) => {
      e.preventDefault();
      const input = form.querySelector<HTMLInputElement>('input[type="email"]');
      const submit = form.querySelector<HTMLButtonElement>('button[type="submit"]');
      if (input && submit) {
        submit.innerHTML = '<span>You\'re in! ✓</span>';
        submit.disabled = true;
        input.value = '';
      }
    });
  }

  // ── Steps section: scroll-driven horizontal track ──────────────────────────
  const scroller = document.querySelector<HTMLElement>('[data-steps-scroller]');
  const track = document.querySelector<HTMLElement>('[data-steps-track]');
  const bar = document.querySelector<HTMLElement>('[data-steps-bar]');
  const dots = Array.from(document.querySelectorAll<HTMLButtonElement>('.steps-progress__dot'));
  const cards = Array.from(document.querySelectorAll<HTMLElement>('.step-card'));

  if (scroller && track && cards.length > 0) {
    const total = cards.length;
    const isDesktop = () => window.matchMedia('(min-width: 769px)').matches;

    let ticking = false;
    const setActive = (idx: number) => {
      cards.forEach((c, i) => c.classList.toggle('is-active', i === idx));
      dots.forEach((d, i) => d.classList.toggle('is-active', i === idx));
    };

    const updateSteps = () => {
      if (!isDesktop()) {
        // Mobile: clear desktop track transform, let the native carousel drive it
        track.style.removeProperty('--steps-x');
        // Sync active card with horizontal scroll position of the track
        const cardWidth = track.clientWidth;
        const idx = cardWidth > 0
          ? Math.min(total - 1, Math.max(0, Math.round(track.scrollLeft / cardWidth)))
          : 0;
        setActive(idx);
        if (bar) bar.style.width = `${(idx / Math.max(1, total - 1)) * 100}%`;
        ticking = false;
        return;
      }

      const rect = scroller.getBoundingClientRect();
      const totalScroll = rect.height - window.innerHeight;
      const scrolled = Math.min(1, Math.max(0, -rect.top / Math.max(1, totalScroll)));
      const stepFloat = scrolled * (total - 1);
      const activeIdx = Math.min(total - 1, Math.round(stepFloat));

      track.style.setProperty('--steps-x', `-${stepFloat * 100}%`);
      setActive(activeIdx);
      if (bar) bar.style.width = `${scrolled * 100}%`;
      ticking = false;
    };

    window.addEventListener('scroll', () => {
      if (!ticking) {
        requestAnimationFrame(updateSteps);
        ticking = true;
      }
    }, { passive: true });
    window.addEventListener('resize', updateSteps, { passive: true });
    // Mobile: track's own horizontal scroll updates the active state
    track.addEventListener('scroll', () => {
      if (!ticking) {
        requestAnimationFrame(updateSteps);
        ticking = true;
      }
    }, { passive: true });

    cards[0].classList.add('is-active');
    updateSteps();

    // Click a dot to jump
    dots.forEach((dot, i) => {
      dot.addEventListener('click', () => {
        if (isDesktop()) {
          const rect = scroller.getBoundingClientRect();
          const totalScroll = rect.height - window.innerHeight;
          const targetScroll = (i / Math.max(1, total - 1)) * totalScroll;
          const sectionTop = rect.top + window.scrollY;
          window.scrollTo({ top: sectionTop + targetScroll, behavior: 'smooth' });
        } else {
          // Mobile: scroll the track horizontally
          track.scrollTo({ left: i * track.clientWidth, behavior: 'smooth' });
        }
      });
    });
  }
</script>
